ESP32 IoT Project - Version 1

#include <WiFi.h>
#include <HTTPClient.h>
#include "DHT.h"

#define WIFI_SSID "IOT LAB"
#define WIFI_PASSWORD "iotlab@svcet"
#define SERVER_URL "https://console.thingzmate.com/api/v1/device-types/3542/devices/exp-3/uplink"
#define DOWNLINK_URL "https://console.thingzmate.com/api/v1/device-types/3542/devices/exp-3/downlink"
#define AUTH_TOKEN "Bearer 1aa521bb0add1a5e3366ed44410885c8"

#define DHTPIN 26
#define DHTTYPE DHT11
#define LED_PIN 2

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(115200);
  dht.begin();
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);

  Serial.print("Connecting to WiFi");
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected!");
}

void loop() {
  float temp = dht.readTemperature();
  float hum = dht.readHumidity();

  if (isnan(temp) || isnan(hum)) {
    Serial.println("Error reading sensor");
  } else {
    Serial.print("Temp: ");
    Serial.print(temp);
    Serial.print(" C, Humidity: ");
    Serial.println(hum);

    String payload = "{\"temperature\":" + String(temp) + ",\"humidity\":" + String(hum) + "}";

    HTTPClient http;
    http.begin(SERVER_URL);
    http.addHeader("Content-Type", "application/json");
    http.addHeader("Authorization", AUTH_TOKEN);

    int code = http.POST(payload);
    Serial.print("Sent data, response code: ");
    Serial.println(code);
    http.end();
  }

  HTTPClient http;
  http.begin(DOWNLINK_URL);
  http.addHeader("Authorization", AUTH_TOKEN);
  int code = http.GET();

  if (code > 0) {
    String resp = http.getString();
    Serial.print("Downlink: ");
    Serial.println(resp);

    if (resp.indexOf("1") != -1) {
      digitalWrite(LED_PIN, HIGH);
      Serial.println("LED ON");
    } else if (resp.indexOf("0") != -1) {
      digitalWrite(LED_PIN, LOW);
      Serial.println("LED OFF");
    }
  } else {
    Serial.print("Downlink error: ");
    Serial.println(code);
  }
  http.end();

  delay(60000);  // wait 1 minute
}

ESP32 IoT Project - Version 2

#include <WiFi.h>
#include <HTTPClient.h>
#include "DHT.h"

#define WIFI_SSID "IOT LAB"
#define WIFI_PASSWORD "iotlab@svcet"
#define SERVER_URL "https://console.thingzmate.com/api/v1/device-types/3542/devices/exp-3/uplink"
#define DOWNLINK_URL "https://console.thingzmate.com/api/v1/device-types/3542/devices/exp-3/downlink"
#define AUTH_TOKEN "Bearer 1aa521bb0add1a5e3366ed44410885c8"

#define DHTPIN 26
#define DHTTYPE DHT11
#define LED_PIN 2

DHT dht(DHTPIN, DHTTYPE);

void setup() {   
  Serial.begin(115200);
  dht.begin();
  pinMode(LED_PIN, OUTPUT);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected!");
}

void loop() {
  float t = dht.readTemperature();
  float h = dht.readHumidity();

  if (!isnan(t) && !isnan(h)) {
    Serial.printf("Temp: %.1f C, Humidity: %.1f\n", t, h);

    HTTPClient http;
    http.begin(SERVER_URL);
    http.addHeader("Content-Type", "application/json");
    http.addHeader("Authorization", AUTH_TOKEN);
    String payload = "{\"temperature\":" + String(t) + ",\"humidity\":" + String(h) + "}";
    int res = http.POST(payload);
    Serial.printf("Sent data, response code: %d\n", res);
    http.end();
  } else {
    Serial.println("Error reading sensor");
  }

  HTTPClient http;
  http.begin(DOWNLINK_URL);
  http.addHeader("Authorization", AUTH_TOKEN);
  int res = http.GET();

  if (res > 0) {
    String r = http.getString();
    Serial.println("Downlink: " + r);
    digitalWrite(LED_PIN, r.indexOf("1") != -1 ? HIGH : LOW);
    Serial.println(r.indexOf("1") != -1 ? "LED ON" : "LED OFF");
  } else {
    Serial.printf("Downlink error: %d\n", res);
  }
  http.end();

  delay(60000); // Wait 1 minute
}
